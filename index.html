body { background: #111; color: #fff; text-align: center; font-family: Arial, sans-serif; } canvas { background: #000; display: block; margin: 20px auto; border: 3px solid #fff; } const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); // Paddle const paddleHeight = 100; const paddleWidth = 10; // Player paddle let playerY = canvas.height / 2 - paddleHeight / 2; // AI paddle let aiY = canvas.height / 2 - paddleHeight / 2; // Ball let ballX = canvas.width / 2; let ballY = canvas.height / 2; let ballSize = 10; let ballSpeedX = 5; let ballSpeedY = 4; // Draw rectangle function drawRect(x, y, w, h, color) { ctx.fillStyle = color; ctx.fillRect(x, y, w, h); } // Draw ball function drawBall() { ctx.fillStyle = "white"; ctx.fillRect(ballX, ballY, ballSize, ballSize); } // Game loop function update() { // Move ball ballX += ballSpeedX; ballY += ballSpeedY; // Ball bounce off top/bottom if (ballY <= 0 || ballY + ballSize >= canvas.height) { ballSpeedY = -ballSpeedY; } // Ball bounce off player paddle if (ballX <= paddleWidth && ballY + ballSize >= playerY && ballY <= playerY + paddleHeight) { ballSpeedX = -ballSpeedX; } // Ball bounce off AI paddle if (ballX + ballSize >= canvas.width - paddleWidth && ballY + ballSize >= aiY && ballY <= aiY + paddleHeight) { ballSpeedX = -ballSpeedX; } // Reset ball if missed if (ballX < 0 || ballX > canvas.width) { ballX = canvas.width / 2; ballY = canvas.height / 2; } // AI paddle follows ball if (aiY + paddleHeight / 2 < ballY) aiY += 4; if (aiY + paddleHeight / 2 > ballY) aiY -= 4; } // Draw everything function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Player paddle drawRect(0, playerY, paddleWidth, paddleHeight, "white"); // AI paddle drawRect(canvas.width - paddleWidth, aiY, paddleWidth, paddleHeight, "white"); // Ball drawBall(); } // Loop function gameLoop() { update(); draw(); requestAnimationFrame(gameLoop); } gameLoop(); // Controls document.addEventListener("mousemove", (e) => { const rect = canvas.getBoundingClientRect(); playerY = e.clientY - rect.top - paddleHeight / 2; });
